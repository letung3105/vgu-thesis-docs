\section{Results}

\begin{frame}{Evaluation metric}
    \begin{block}{Mean absolute percentage error}
        \begin{equation*}
            MAPE = \frac{100}{n} \sum_{i=1}^n \left| \frac{\hat{y}_i - y_i}{y_i} \right|
        \end{equation*}
    \end{block}
\end{frame}
